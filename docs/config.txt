  env:           "local"/"dev"/"prod"

  logger:
    type:        "base" - обязательное поле
    appLevel:    "Debug"/"Info"/"Error"/"Off"
    entLevel:    "Debug"/"Info"/"Error"/"Off"
    simLevel:    "Debug"/"Info"/"Error"/"Off"
    appPath:     строка - адрес сохранения логов приложения
    entPath:     строка - адрес сохранения логов сущностей
    simPath:     строка - адрес сохранения логов симуляции

  modelConst:
    maxGen:      число - максимальный размер гена
    foodPercent: число - целый процент количества еды в мире
    lengthDNA:   число - длинна генома
    energyPoint: число - награда за поедание еды
    brain:       "16/64" - тип интерпретатора генома:
    loop:        зацикливание мира по:
        x:       булево значение - координата x
        y:       булево значение - координата y

  simulation:
    type:        "train" - обязательное поле
    X:           число - размер мира по x
    Y:           число - размер мира по y
    startPop:    число - размер стартовой популяции
    endPop:      число - размер конечной популяции, должно равняться квадратному корню от startPop
    startPoison: число - стартовый уровень яда в каждой клетке мира
    resourceUpdate: число - частота обновления ресурсов в мире
    ageExit:     число - возраст выхода из мира
    mutation:    число - количество мутаций при переходе в другой мир

    console:
        type:    "old" - обязательное поле
        timeOut: число - время задержки вывода кадра

  storage:
    type:        "file" - обязательное поле
    pathAge:     строка - адрес сохранения возраста мира
    pathTrain:   строка - адрес сохранения итогов обучения
